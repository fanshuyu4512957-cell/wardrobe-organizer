  nt.js"></script>

      <script src="https://cdn.jsdelivr.net/npm/react-dom@18.0.0/umd/react-dom.d

  evelopment.js"></script>

      <script 

  src="https://cdn.jsdelivr.net/npm/@babel/standalone/babel.js"></script>

      <script src="https://cdn.tailwindcss.com"></script>

      <link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/font-a

  wesome/5.15.3/css/all.min.css">

  </head>

      <div id="root"></div>


      <script type="text/babel">

          const { useState } = React;


          // 模拟AI识别

          const simulateAIService = (imageFile) => {

              return new Promise((resolve) => {

                  setTimeout(() => {

                      const mockResults = [

                          { type: '上衣', count: 5, confidence: 0.95 },

                          { type: '裤子', count: 3, confidence: 0.92 },

                          { type: '裙子', count: 2, confidence: 0.88 },

                          { type: '外套', count: 1, confidence: 0.85 },

                      ];

                      resolve(mockResults);

                  }, 2000);

              });

          };


          // 模拟规划服务

          const simulatePlanningService = (clothingItems) => {

              return new Promise((resolve) => {

                  setTimeout(() => {

                      const plans = [

                          {

                              space: { id: 'drawer1', name: '抽屉1', capacity:

  20, type: 'drawer', location: '左侧' },

                              items: clothingItems.filter(item => item.type ===

  '上衣' || item.type === '裤子'),

                              utilization: 0.7

                          },

                          {

                              space: { id: 'hanger1', name: '挂杆1', capacity:

  15, type: 'hanger', location: '中间' },

                              items: clothingItems.filter(item => item.type ===

  '裙子' || item.type === '外套'),

                              utilization: 0.6

                          }

                      ];

                      resolve(plans);

                  }, 1000);

              });

          };


          function App() {

              const [clothingItems, setClothingItems] = useState([

                  { type: '上衣', count: 5, confidence: 0.95 },

                  { type: '裤子', count: 3, confidence: 0.92 },

              ]);

              const [plans, setPlans] = useState([]);

              const [selectedImage, setSelectedImage] = useState(null);

              const [isAnalyzing, setIsAnalyzing] = useState(false);

              const [isPlanning, setIsPlanning] = useState(false);


              const handleImageUpload = async (event) => {

                  const file = event.target.files[0];

                  if (file) {

                      setSelectedImage(URL.createObjectURL(file));

                      setIsAnalyzing(true);


                      try {

                          const results = await simulateAIService(file);

                          setClothingItems(results);

                          alert(`识别完成！检测到: ${results.map(item =>

  `${item.type} x${item.count}`).join(', ')}`);

                      } catch (error) {

                          alert('识别失败，请重试');

                      } finally {

                          setIsAnalyzing(false);

                      }

                  }

              };


              const handleGeneratePlan = async () => {

                  if (clothingItems.length === 0) {

                      alert('请先添加或识别衣物');

                      return;

                  }


                  setIsPlanning(true);


                  try {

                      const generatedPlans = await

  simulatePlanningService(clothingItems);

                      setPlans(generatedPlans);

                  } catch (error) {

                      alert('规划失败，请重试');

                  } finally {

                      setIsPlanning(false);

                  }

              };


              const handleAddItem = (item) => {

                  setClothingItems([...clothingItems, item]);

              };


              const handleEditItem = (updatedItem) => {

                  setClothingItems(clothingItems.map(item =>

                      item.type === updatedItem.type ? updatedItem : item

                  ));

              };


              const handleDeleteItem = (type) => {

                  setClothingItems(clothingItems.filter(item => item.type !==

  type));

              };


              return (

                  <div className="min-h-screen bg-gray-50">

                      {/* 导航栏 */}

                      <nav className="bg-blue-600 text-white p-4">

                          <div className="container mx-auto flex justify-between

   items-center">

                              <h1 className="text-2xl

  font-bold">衣柜收纳规划</h1>

                              <div className="flex space-x-4">

                                  <button className="px-4 py-2 bg-blue-700

  rounded hover:bg-blue-800">

                                      拍照识别

                                  </button>

                                  <button className="px-4 py-2 bg-blue-700

  rounded hover:bg-blue-800">

                                      衣物管理

                                  </button>

                                  <button className="px-4 py-2 bg-blue-700

  rounded hover:bg-blue-800">

                                      生成规划

                                  </button>

                              </div>

                          </div>

                      </nav>


                      <div className="container mx-auto p-4">

                          {/* 拍照识别区域 */}

                          <div className="bg-white rounded-lg shadow-md p-6

  mb-6">

                              <h2 className="text-xl font-semibold

  mb-4">拍照识别衣物</h2>

                              <div className="flex flex-col items-center">

                                  <input

                                      type="file"

                                      accept="image/*"

                                      onChange={handleImageUpload}

                                      className="mb-4"

                                  />

                                  {selectedImage && (

                                      <img src={selectedImage} alt="上传的图片" 

  className="max-w-full h-64 object-cover rounded" />

                                  )}

                                  {isAnalyzing && (

                                      <div className="text-center">

                                          <div className="inline-block

  animate-spin rounded-full h-8 w-8 border-b-2 border-blue-600"></div>

                                          <p className="mt-2">正在识别...</p>

                                      </div>

                                  )}

                              </div>

                          </div>


                          {/* 衣物管理区域 */}

                          <div className="bg-white rounded-lg shadow-md p-6

  mb-6">

                              <h2 className="text-xl font-semibold

  mb-4">衣物管理</h2>

                              <div className="mb-4">

                                  <button

                                      onClick={handleGeneratePlan}

                                      disabled={isPlanning}

                                      className="px-4 py-2 bg-green-600

  text-white rounded hover:bg-green-700 disabled:bg-gray-400"

                                  >

                                      {isPlanning ? '生成中...' :

  '生成收纳规划'}

                                  </button>

                              </div>

                              <div className="space-y-2">

                                  {clothingItems.map((item, index) => (

                                      <div key={index} className="border p-3

  rounded">

                                          <div className="flex justify-between

  items-center">

                                              <span>{item.type}</span>

                                              <span>数量: {item.count}</span>

                                              <span>置信度: {(item.confidence *

  100).toFixed(1)}%</span>

                                          </div>

                                      </div>

                                  ))}

                              </div>

                          </div>


                          {/* 规划结果区域 */}

                          {plans.length > 0 && (

                              <div className="bg-white rounded-lg shadow-md

  p-6">

                                  <h2 className="text-xl font-semibold

  mb-4">收纳规划结果</h2>

                                  <div className="space-y-4">

                                      {plans.map((plan, index) => (

                                          <div key={index} className="border p-4

   rounded">

                                              <h3 

  className="font-semibold">{plan.space.name}</h3>

                                              <p>位置: {plan.space.location}</p>

                                              <p>类型: {plan.space.type}</p>

                                              <p>容量:

  {plan.space.capacity}件</p>

                                              <p>利用率: {(plan.utilization *

  100).toFixed(1)}%</p>

                                              <div className="mt-2">

                                                  <p 

  className="font-semibold">存放物品:</p>

                                                  {plan.items.map((item,

  itemIndex) => (

                                                      <p key={itemIndex}>

                                                          - {item.type}:

  {item.count}件

                                                      </p>

                                                  ))}

                                              </div>

                                          </div>

                                      ))}

                                  </div>

                              </div>

                          )}

                      </div>

                  </div>

              );

          }


          ReactDOM.render(<App />, document.getElementById('root'));

      </script>
